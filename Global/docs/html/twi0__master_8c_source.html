47bool isTWIBad(void) {
48 //Checks for: NACK, ARBLOST, BUSERR, Bus Busy
49 if ((((TWI0.MSTATUS) & (TWI_RXACK_bm | TWI_ARBLOST_bm | TWI_BUSERR_bm)) > 0) || (TWI_IS_BUSBUSY())) {
50 return true;
51 }
52 return false;
53}

55void TWI_initHost(void) {
56 //Standard 100kHz TWI, 4 Cycle Hold, 50ns SDA Hold Time
57 TWI0.CTRLA = TWI_SDAHOLD_50NS_gc;
58
59 //Clear Dual Control
60 TWI0.DUALCTRL = 0x00;
61
62 //Enable Run in Debug
63 TWI0.DBGCTRL = TWI_DBGRUN_bm;
64
65 //Clear MSTATUS (write 1 to flags). BUSSTATE set to idle

69 //Set for 100kHz from a 4MHz oscillator

91bool _startTWI(uint8_t addr, bool read) {
92 //If the Bus is Busy
93 if (TWI_IS_BUSBUSY()) {
94 return false;
95 }
96
97 //Send Address
98 TWI0.MADDR = (addr << 1) | read;
99
100 //Wait...
101 TWI_WAIT();
102
103 if (isTWIBad()) {
104 //Stop the Bus
105 TWI0.MCTRLB = TWI_MCMD_STOP_gc;
106 return false;
107 }
108
109 //TWI Started
110 return true;

113//Internal Function: Reads len bytes from TWI, then issues a bus STOP
114void _readFromTWI(uint8_t* data, uint8_t len) {
115 uint8_t bCount = 0;
116
117 //Release the clock hold
118
119 TWI0.MSTATUS = TWI_CLKHOLD_bm;
120
121 //TWI0.MCTRLB = TWI_MCMD_RECVTRANS_gc;
122
123 while (bCount < len) {
124 //Wait...
125 TWI_WAIT();
126
127 //Store data
128 data[bCount] = TWI0.MDATA;
129
130 //Increment the counter
131 bCount += 1;
132
133 if (bCount != len) {
134 //If not done, then ACK and read the next byte
135 TWI0.MCTRLB = TWI_ACKACT_ACK_gc | TWI_MCMD_RECVTRANS_gc;
136 }
137 }
138
139 //NACK and STOP the bus
140 TWI0.MCTRLB = TWI_ACKACT_NACK_gc | TWI_MCMD_STOP_gc;
141}

143//Internal Function: Write len bytes to TWI. Does NOT STOP the bus. Returns true if successful
144bool _writeToTWI(uint8_t* data, uint8_t len) {
145 uint8_t count = 0;
146
147 while (count < len) {
148 //Write a byte
149 TWI0.MDATA = data[count];
150
151 //Wait...
152 TWI_WAIT();
153
154 //If the client NACKed, then abort the write
155 if (CLIENT_NACK()) {
156 return false;
157 }
158
159 //Increment the counter
160 count++;
161 }
162
163 return true;
164}

220bool TWI_sendAndReadBytes(uint8_t addr, uint8_t regAddress, uint8_t* data, uint8_t len) {
221 //Address Client Device (Write)
222 if (!_startTWI(addr, TWI_WRITE)) {
223 return false;
224 }
225
226 //Write register address
227 if (!_writeToTWI(&regAddress, 1)) {
228 TWI0.MCTRLB = TWI_MCMD_STOP_gc;
229 return false;
230 }
231
232 //Restart the TWI Bus in READ mode
233 TWI0.MADDR |= TWI_READ;
234 TWI0.MCTRLB = TWI_MCMD_REPSTART_gc;
235
236 //Wait...
237 TWI_WAIT();
238
239 if (isTWIBad()) {
240 //Stop the TWI Bus if an error occurred
241 TWI0.MCTRLB = TWI_MCMD_STOP_gc;
242 return false;
243 }
244
245 //Read the data from the client
246 _readFromTWI(data, len);
247
248 return true;
